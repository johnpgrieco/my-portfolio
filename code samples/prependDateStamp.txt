function prependDateStamp() {
  
  // Prepend the date stamp to every file in a Google Drive.
  // Toggle the first two lines based on test or production 
  // TEST: var files = DriveApp.getFilesByName('THE TEST FILES TO MANIPULATE');
  // PRODUCTION, GET ALL THE FILES IN THE DRIVE: var files = DriveApp.getFiles();
  
  // So, it is currently set up to only rename files entitled "Untitled Spreadsheet To Test.xlsx"....

  // Steps to deploy this: 
  // 1. Unit test: run it on a local drive. DONE.
  // 2. Back up the entire production drive. This will likely change the datestamp
  //    in the event of a need for rollback, but protects from data loss. DONE. 
  // 3. QA: Run this on the test folder/files in production. DONE. 
  // 4. Prepare: uncomment the statement var files = DriveApp.getFiles() below
  // 5. Prepare: comment out the statement var files = DriveApp.getFilesByName below
  // 6. Prepare: also consider commenting out the audit trail logging to conserve space.  
  // 7. Save this script to the production environment. Refer to https://developers.google.com/apps-script/guides/projects
  // 8. Run it on the entire production drive. 

  var files = DriveApp.getFilesByName('Untitled Spreadsheet To Test.xlsx');
  var fileCount = 0; 
  //var files = DriveApp.getFiles(); 
  while (files.hasNext()) { 
    var file = files.next(); 
    oldFileName = file.getName();
    
    Logger.log('The old file name is: ' + oldFileName);
    
    var fileYear = file.getLastUpdated().getFullYear(); 
    var fileMonth = file.getLastUpdated().getMonth() + 1;   //getMonth() returns 0 for January, 1 for February, etc. so add 1. 
    var fileDayOfMonth = file.getLastUpdated().getDate(); 
    
    //Format it nicely
    if (fileMonth < 10) { 
      fileMonth = '0' + fileMonth; 
    }
    if (fileDayOfMonth < 10) { 
      fileDayOfMonth = '0' + fileDayOfMonth; 
    } 
    //Logger.log('The year is: ' + fileYear);
    //Logger.log('The month is : ' + fileMonth);  
    //Logger.log('the day of month is: ' + fileDayOfMonth); 
    
    var newFileName = '' + fileYear + fileMonth + fileDayOfMonth + '-' + oldFileName;
    file.setName(newFileName);
    
    Logger.log('The new file name is: ' + newFileName); 
    fileCount++;
  } 
  
  Logger.log('The count of files renamed is: ' + fileCount);
